# TypeScript

# 基础类型

# 布尔值

布尔值就是简单的true/false值

let isDone: boolean = false;

# 数字

和JavaScript一样，TypeScript里所有数字都是浮点数。浮点数类型为number。除了支持十进制和十六进制字面量，TypeScript还支持ECMAScript 2015中引入的二进制和八进制字面量。

let hexLiteral: number = 0xf00d;
let binaryLiteral: number = 0b1010;
let octalLiteral: number = 0o744;
let decLiteral: number = 6;

# 字符串

let name: string = "bob";

# 数组

有两种方法来定义数组

  1. 元素类型+[]

    let list: number[] = [1, 2, 3]

  2. Array<元素类型>

    let list: Array<number> = [1, 2, 3]

# 元组

用于表示已知元素数量和类型的数组，各元素的类型不必相同

let x: [string, number]

x = ['hello', 10] //OK

x = [10, 'hellow'] //Error
